#include <iostream>
using namespace std;

int main(void)
{
	int *pnum;
	int num = 6;
	pnum = &num;
	cout << pnum;
}

void insert(int idx, int num, int arr[], int& len) {
	for (int i=len; i > idx; i--) {
		arr[i] = arr[i-1];
	};
	arr[idx] = num;
	len++;
}

void erase(int idx, int arr[], int& len) {
	len--;
	for (int i=idx; i < len; i++) {
		arr[i] = arr[i+1];
	};
}

void printArr(int arr[], int len) {
	for (int i=0; i<len; i++) {
		cout << arr[i] << ' ';
	};
	cout << '\n';
}

int main() {
	int arr[10] = {1, 2, 3, 4, 5, 6};
	int len = 6;
	insert(3, 60, arr, len);
	printArr(arr, len);
	erase(4, arr, len);
	printArr(arr, len);
	
}
