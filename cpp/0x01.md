int func2(int arr[], int N) {
	int i, arrCounting[101] = {0};
	for (i=0; i<N; i++) {
		arrCounting[arr[i]] += 1;
	}
	if (arrCounting[50] >= 2) {
		return 1;
	}
	for (i=0; i<50; i++) {
		if (arrCounting[i] && arrCounting[100-i]) {
			return 1;
		}
	}
	return 0;
}

int main2() {
	int numArr[3] = {50, 50, 30};
	int numArr2[3] = {1,2, 99};
	cout << func2(numArr, 3) << endl <<func2(numArr2, 3)<< endl;
